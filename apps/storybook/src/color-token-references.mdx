import { token } from 'styled-system/tokens'

export const isSemanticLeaf = (node) =>
  typeof node?.value?.base === 'string' && typeof node?.value?._dark === 'string'

export const flattenSemanticColorRows = (node, path = []) => {
  if (isSemanticLeaf(node)) {
    const normalizedPath = path.filter((segment) => segment !== 'DEFAULT');
    return [
      {
        tokenPath: `colors.${normalizedPath.join('.')}`,
        base: node.value.base,
        dark: node.value._dark,
      },
    ];
  }

if (!node || typeof node !== 'object') {
return [];
}

return Object.entries(node).flatMap(([key, value]) =>
flattenSemanticColorRows(value, [...path, key]),
);
};

export const semanticColorRows = flattenSemanticColorRows(semanticTokens.colors)

export const getSemanticRows = (...prefixes) =>
  semanticColorRows.filter((row) => prefixes.some((prefix) => row.tokenPath.startsWith(prefix)))

export const orderRows = (rows, order) =>
  rows
    .map((row, index) => ({ index, rank: order.indexOf(row.tokenPath), row }))
    .sort((a, b) => {
      if (a.rank === -1 && b.rank === -1) return a.index - b.index
      if (a.rank === -1) return 1
      if (b.rank === -1) return -1
      return a.rank - b.rank
    })
    .map(({ row }) => row)
